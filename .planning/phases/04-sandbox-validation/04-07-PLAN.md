```
"""SAFE-05 graceful-shutdown subprocess test (Scenario 9: real SIGINT against running server)."""
from __future__ import annotations

import os
import sys
import json
import time
import socket
import signal
import asyncio
import subprocess
import urllib.request

import pytest
import structlog

from arbiter.sandbox import evidence

log = structlog.get_logger("arbiter.sandbox.shutdown")

# Operator-supplied from Task 0.
SHUTDOWN_MARKET_TICKER = "REPLACE-WITH-OPERATOR-SUPPLIED-TICKER"
SHUTDOWN_PRICE = 0.05
SHUTDOWN_QTY = 3


def _free_port() -> int:
    """Pattern from arbiter/test_api_integration.py:14-21."""
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as sock:
        try:
            sock.bind(("127.0.0.1", 0))
        except PermissionError as exc:
            pytest.skip(f"Local socket bind unavailable: {exc}")
        return sock.getsockname()[1]


def _wait_for_server(port: int, timeout: float = 20.0) -> None:
    """Pattern from arbiter/test_api_integration.py:24-34."""
    start = time.time()
    while time.time() - start < timeout:
        try:
            with urllib.request.urlopen(f"http://127.0.0.1:{port}/api/health", timeout=1) as response:
                if response.status == 200:
                    return
        except Exception:
            time.sleep(0.25)
    raise AssertionError(f"Server on port {port} did not become ready within {timeout}s")


@pytest.mark.live
async def test_sigint_cancels_open_kalshi_demo_orders(evidence_dir):
    """Launch arbiter.main as subprocess, place resting Kalshi demo order, SIGINT, verify graceful shutdown."""
    assert "arbiter_sandbox" in os.getenv("DATABASE_URL", ""), "wrong DB"
    assert "demo-api.kalshi.co" in os.getenv("KALSHI_BASE_URL", ""), "wrong Kalshi URL"

    port = _free_port()

    # Build subprocess env — inherit current env (which sourced .env.sandbox) + explicit overrides.
    env = dict(os.environ)
    env["PHASE4_MAX_ORDER_USD"] = os.getenv("PHASE4_MAX_ORDER_USD", "5")  # sanity check present

    # Capture stderr into evidence_dir/run.log.jsonl.
    stderr_path = evidence_dir / "run.log.jsonl"
    stderr_fh = open(stderr_path, "w", encoding="utf-8")

    # Windows note: on Windows, SIGINT delivery to a child requires CREATE_NEW_PROCESS_GROUP + CTRL_BREAK_EVENT.
    creationflags = 0
    if sys.platform == "win32":
        creationflags = subprocess.CREATE_NEW_PROCESS_GROUP

    proc = subprocess.Popen(
        [sys.executable, "-m", "arbiter.main", "--api-only", "--port", str(port)],
        cwd=os.getcwd(),
        env=env,
        stdout=subprocess.DEVNULL,
        stderr=stderr_fh,
        creationflags=creationflags,
    )

    client_order_id = None
    shutdown_events_captured = []
    return_code = None

    try:
        _wait_for_server(port)
        log.info("scenario.shutdown.server_ready", port=port, pid=proc.pid)

        # Place a resting Kalshi demo order using our own in-test adapter — NOT via the subprocess's
        # API. The subprocess is the server being shutdown-tested; our test harness places the order.
        # Alternative: POST to subprocess's /api/place-order if the API exposes it — simpler but assumes
        # that endpoint exists. Prefer in-test adapter to keep coupling minimal.
        from arbiter.config.settings import load_config
        from arbiter.execution.adapters.kalshi import KalshiAdapter

        cfg = load_config()
        adapter = KalshiAdapter(cfg)
        client_order_id = f"ARB-SANDBOX-SHUTDOWN-YES-{os.urandom(4).hex()}"

        # Place non-FOK resting order (see Plan 04-03 Task 3 for method-name caveat).
        placed = await adapter.place_resting_limit(
            arb_id="ARB-SANDBOX-SHUTDOWN",
            market_id=SHUTDOWN_MARKET_TICKER,
            canonical_id=SHUTDOWN_MARKET_TICKER,
            side="yes",
            price=SHUTDOWN_PRICE,
            qty=SHUTDOWN_QTY,
            client_order_id=client_order_id,
        )
        log.info("scenario.shutdown.order_placed", order_id=placed.order_id, client_order_id=client_order_id)

        # PROBLEM: the order was placed against Kalshi demo but the subprocess running arbiter.main is a
        # separate process. For SAFE-05 to cancel it on shutdown, the subprocess must know about the order.
        #
        # Options:
        # (A) Have the subprocess place the order via its own API endpoint (requires POST /api/place-order
        #     or similar — inspect api.py for what's exposed).
        # (B) Recognize that SAFE-05's cancel_all path (inside the subprocess's KalshiAdapter) calls the
        #     Kalshi API GET /portfolio/orders to discover ALL open orders for the account, not just ones
        #     it tracked locally. If that is how cancel_all is implemented (per 03-05-PLAN.md), then an
        #     order placed by our test harness IS visible to the subprocess's adapter on shutdown cancel.
        #
        # Option (B) is more robust and aligns with SAFE-05's "cancel EVERYTHING" semantic. Confirm by
        # reading arbiter/execution/adapters/kalshi.py::cancel_all — does it list server-side orders
        # or only track orders it placed?
        #
        # ASSUMPTION: cancel_all uses server-side listing. If confirmed, proceed as below. If not,
        # switch to option (A) and call POST /api/place-order against the subprocess.

        await asyncio.sleep(1.0)  # allow order to rest

        # Send SIGINT.
        log.info("scenario.shutdown.sending_sigint", pid=proc.pid)
        if sys.platform == "win32":
            proc.send_signal(signal.CTRL_BREAK_EVENT)
        else:
            os.kill(proc.pid, signal.SIGINT)

        # Wait for subprocess to exit.
        try:
            return_code = proc.wait(timeout=20.0)
        except subprocess.TimeoutExpired:
            log.warning("scenario.shutdown.timeout_waiting_for_exit")
            proc.kill()
            return_code = proc.wait(timeout=5.0)
            pytest.fail("Subprocess did not exit within 20s of SIGINT — SAFE-05 deadlock risk")

        log.info("scenario.shutdown.exited", return_code=return_code)

        # Close stderr handle and parse JSONL.
        stderr_fh.close()
        with open(stderr_path, "r", encoding="utf-8") as fh:
            for line in fh:
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                except json.JSONDecodeError:
                    continue
                event_type = obj.get("type") or obj.get("event")
                if event_type in ("shutdown_state",) or "shutdown" in str(event_type or "").lower():
                    shutdown_events_captured.append(obj)

        log.info("scenario.shutdown.events_found", count=len(shutdown_events_captured))

        # Assertion 1: shutdown phase sequence appears in stderr.
        phases_seen = {e.get("payload", {}).get("phase") or e.get("phase") for e in shutdown_events_captured}
        assert "shutting_down" in phases_seen or "shutting_down" in str(shutdown_events_captured), (
            f"SAFE-05 INVARIANT: expected 'shutting_down' phase in stderr; got phases={phases_seen} "
            f"events={shutdown_events_captured[:3]}"
        )
        assert "complete" in phases_seen or "complete" in str(shutdown_events_captured), (
            f"SAFE-05 INVARIANT: expected 'complete' phase in stderr; got phases={phases_seen}"
        )

        # Assertion 2: the order is CANCELLED on the demo exchange.
        await asyncio.sleep(1.0)  # allow platform to finalize cancel
        remaining = await adapter.list_open_orders_by_client_id(client_order_id)
        assert not remaining, (
            f"SAFE-05 INVARIANT VIOLATED: order {client_order_id} still open on Kalshi demo after subprocess SIGINT. "
            f"cancel_all path did not cancel it. remaining={remaining}"
        )

        # Assertion 3: subprocess exited cleanly.
        # Project convention for clean-shutdown return code — Phase 3 may have specific behavior.
        # Accept 0 (unix convention) or any exit != forced-kill (-9).
        assert return_code is not None and return_code >= 0, (
            f"Subprocess exited with {return_code} — forced kill or crash"
        )

    finally:
        if not stderr_fh.closed:
            stderr_fh.close()
        if proc.poll() is None:
            proc.kill()
            try:
                proc.wait(timeout=5)
            except subprocess.TimeoutExpired:
                pass
        # Cleanup adapter if built
        try:
            close = getattr(adapter, "close", None) or getattr(adapter, "aclose", None)
            if close is not None:
                r = close()
                if hasattr(r, "__await__"):
                    await r
        except Exception:
            pass

    (evidence_dir / "scenario_manifest.json").write_text(json.dumps({
        "scenario": "sigint_cancels_open_kalshi_demo_orders",
        "requirement_ids": ["SAFE-05", "TEST-01"],
        "phase_3_refs": ["03-05-PLAN", "03-HUMAN-UAT.md Test 2 (partial — backend only; UI banner reserved)"],
        "tag": "real",
        "subprocess_return_code": return_code,
        "placed_client_order_id": client_order_id,
        "market": SHUTDOWN_MARKET_TICKER,
        "price": SHUTDOWN_PRICE,
        "qty": SHUTDOWN_QTY,
        "shutdown_events_captured_count": len(shutdown_events_captured),
        "phases_seen": list({e.get("payload", {}).get("phase") or e.get("phase") for e in shutdown_events_captured}),
        "order_cancelled_on_platform": True,  # reached end without assertion failure
        "platform": sys.platform,
    }, indent=2), encoding="utf-8")
```
