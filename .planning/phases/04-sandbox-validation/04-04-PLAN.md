```
"""Polymarket real-$1 happy path with fee reconstruction via get_trades (Scenario 2: TEST-02 + TEST-04)."""
from __future__ import annotations

import os
import json
import asyncio
from decimal import Decimal

import pytest
import structlog

from arbiter.sandbox import evidence, reconcile
from arbiter.config.settings import polymarket_order_fee

log = structlog.get_logger("arbiter.sandbox.poly_happy")

# TARGET MARKET — operator supplies in Task 0 response.
HAPPY_TOKEN_ID = "REPLACE-WITH-OPERATOR-SUPPLIED-TOKEN-ID"
HAPPY_PRICE = 0.10
HAPPY_QTY = 10           # notional = 0.10 * 10 = $1.00 — well under $5 hard-lock
HAPPY_CATEGORY = "politics"  # operator-supplied; drives fee rate per Phase 1 D-11


@pytest.mark.live
async def test_polymarket_happy_lifecycle(
    poly_test_adapter, sandbox_db_pool, evidence_dir, balance_snapshot,
):
    """Submit real-$ FOK on Polymarket liquid market -> fill -> USDC delta -> fee reconstruction via get_trades."""
    adapter = poly_test_adapter
    assert "arbiter_sandbox" in os.getenv("DATABASE_URL", ""), "wrong DB"

    # BELT: triple-check notional is safe BEFORE touching adapter.
    notional = HAPPY_PRICE * HAPPY_QTY
    assert notional <= 5.0, (
        f"SAFETY: test notional ${notional:.2f} exceeds $5 limit. "
        f"This would trip the PHASE4_MAX_ORDER_USD hard-lock in the adapter, but we fail-fast here "
        f"so the operator never accidentally sends a $6+ order."
    )

    from arbiter.execution.engine import OrderStatus  # or adapter's import path

    # Pre-flight (Pitfall 5): check min_order_size.
    client = adapter._get_client()
    assert client is not None, "Polymarket client failed to initialize (check POLY_PRIVATE_KEY / signature_type / funder)"

    loop = asyncio.get_event_loop()
    ob = await loop.run_in_executor(None, client.get_order_book, HAPPY_TOKEN_ID)
    min_order_size = float(getattr(ob, "min_order_size", 0))
    log.info("scenario.poly_happy.pre_flight", token=HAPPY_TOKEN_ID, min_order_size=min_order_size, qty=HAPPY_QTY)
    assert HAPPY_QTY >= min_order_size, (
        f"qty={HAPPY_QTY} below market min_order_size={min_order_size} — platform will reject. "
        f"Operator should pick a market with lower min_order_size or raise qty (respecting $5 notional cap)."
    )

    # Pre-balance snapshot (TEST-03).
    pre_balances = await balance_snapshot()

    # Place FOK.
    arb_id = "ARB-SANDBOX-POLY-HAPPY"
    order = await adapter.place_fok(
        arb_id=arb_id,
        market_id=HAPPY_TOKEN_ID,
        canonical_id=HAPPY_TOKEN_ID,
        side="yes",
        price=HAPPY_PRICE,
        qty=HAPPY_QTY,
    )

    log.info(
        "scenario.poly_happy.order_returned",
        order_id=order.order_id, status=str(order.status), error=getattr(order, "error", None),
    )

    # TEST-02: full lifecycle -> FILLED order.
    assert order.status == OrderStatus.FILLED, (
        f"Expected FILLED on Polymarket token {HAPPY_TOKEN_ID}; got {order.status}. "
        f"Error: {getattr(order, 'error', None)}. "
        f"Possible causes: market moved past limit, wallet not funded, SignatureType/funder mismatch, min_order_size not met."
    )

    # TEST-04: reconstruct fee from get_trades (Pitfall 2 — no fee in post_order response).
    # A2 assumption: first run logs trades[0] keys before strict-asserting field names.
    # Per Pattern 4, call get_trades after a brief settle pause.
    await asyncio.sleep(2.0)  # small delay for trade indexing

    try:
        from py_clob_client.clob_types import TradeParams
    except ImportError:
        pytest.fail("py_clob_client.clob_types.TradeParams missing — version mismatch; expected 0.34.6")

    maker_address = getattr(client, "get_address", lambda: os.getenv("POLY_FUNDER"))()
    trades_fn = lambda: client.get_trades(TradeParams(maker_address=maker_address, market=HAPPY_TOKEN_ID))
    trades = await loop.run_in_executor(None, trades_fn)
    log.info("scenario.poly_happy.trades_raw_keys", trades_count=len(trades), first_keys=list(trades[0].keys()) if trades else [])

    # Dump raw trades for A2 verification in SUMMARY.md.
    (evidence_dir / "polymarket_trades_raw.json").write_text(
        json.dumps(trades, indent=2, default=str), encoding="utf-8"
    )

    assert trades, (
        f"get_trades returned empty for maker={maker_address}, market={HAPPY_TOKEN_ID}. "
        f"Possible: trade not yet indexed (raise sleep), wrong maker_address, wrong market param."
    )

    # Reconstruct platform fee. Per Pitfall 2: sum over trades of (fee_rate_bps / 10_000) * size * price * (1 - price).
    # Field names per A2 — adjust if raw-keys log shows different names.
    platform_fee = 0.0
    for trade in trades:
        rate_bps_raw = trade.get("fee_rate_bps", trade.get("feeRateBps", 0))
        size_raw = trade.get("size", trade.get("matched_amount", 0))
        price_raw = trade.get("price", 0)
        rate_bps = float(rate_bps_raw or 0)
        size = float(size_raw or 0)
        price = float(price_raw or 0)
        trade_fee = (rate_bps / 10_000.0) * size * price * (1.0 - price)
        platform_fee += trade_fee
        log.info("scenario.poly_happy.trade_fee", rate_bps=rate_bps, size=size, price=price, fee=trade_fee)

    # Compute predicted fee via local fee function.
    # polymarket_order_fee signature — inspect to confirm argument order + keyword name for category.
    try:
        computed_fee = polymarket_order_fee(HAPPY_PRICE, HAPPY_QTY, category=HAPPY_CATEGORY)
    except TypeError:
        # Fall back to positional if category is not a kwarg.
        computed_fee = polymarket_order_fee(HAPPY_PRICE, HAPPY_QTY)

    reconcile.assert_fee_matches("polymarket", platform_fee, computed_fee)
    log.info("scenario.poly_happy.fee_match", platform_fee=platform_fee, computed_fee=computed_fee)

    # Post-balance snapshot.
    post_balances = await balance_snapshot()

    await evidence.dump_execution_tables(sandbox_db_pool, evidence_dir)
    evidence.write_balances(evidence_dir, pre_balances, post_balances)
    (evidence_dir / "scenario_manifest.json").write_text(json.dumps({
        "scenario": "polymarket_happy_lifecycle",
        "requirement_ids": ["TEST-02", "TEST-04"],
        "tag": "real",
        "order_id": order.order_id,
        "market_token_id": HAPPY_TOKEN_ID,
        "price": HAPPY_PRICE,
        "qty": HAPPY_QTY,
        "notional": notional,
        "category": HAPPY_CATEGORY,
        "platform_fee": platform_fee,
        "computed_fee": computed_fee,
        "status": str(order.status),
        "trades_count": len(trades),
    }, indent=2), encoding="utf-8")
```
