"""Phase 4 acceptance aggregator: collects scenario manifests, applies reconciliation, renders VALIDATION.md."""
from __future__ import annotations

import json
import pathlib
from dataclasses import dataclass, field
from datetime import datetime, timezone
from typing import Any, Dict, List, Optional, Tuple

from arbiter.sandbox.reconcile import RECONCILE_TOLERANCE_USD


DEFAULT_EVIDENCE_ROOT = pathlib.Path("evidence/04")
VALIDATION_MD_PATH = pathlib.Path(".planning/phases/04-sandbox-validation/04-VALIDATION.md")


@dataclass
class ScenarioReconcileResult:
    scenario: str
    evidence_dir: pathlib.Path
    requirement_ids: List[str]
    tag: str  # "real" | "injected"
    pnl_check_applicable: bool
    pnl_delta: Optional[float]
    pnl_recorded: Optional[float]
    pnl_discrepancy: Optional[float]
    pnl_passed: Optional[bool]
    fee_check_applicable: bool
    fee_platform: Optional[float]
    fee_computed: Optional[float]
    fee_discrepancy: Optional[float]
    fee_passed: Optional[bool]
    overall_passed: bool
    notes: str = ""


@dataclass
class ReconcileReport:
    results: List[ScenarioReconcileResult] = field(default_factory=list)
    tolerance: float = RECONCILE_TOLERANCE_USD

    @property
    def any_real_breach(self) -> bool:
        """True if any real-tagged scenario has a reconciliation failure (triggers D-19 hard gate)."""
        return any(
            r.tag == "real" and not r.overall_passed
            for r in self.results
        )

    @property
    def phase_gate_status(self) -> str:
        """PASS / BLOCKED per D-19."""
        return "BLOCKED" if self.any_real_breach else "PASS"


def collect_scenario_manifests(evidence_root: pathlib.Path = DEFAULT_EVIDENCE_ROOT) -> List[Dict[str, Any]]:
    """Glob evidence/04/*/scenario_manifest.json and load each.
    Returns list of dicts; each dict includes an extra key '_evidence_dir' pointing to the parent dir.
    """
    root = pathlib.Path(evidence_root)
    if not root.exists():
        return []
    manifests: List[Dict[str, Any]] = []
    for manifest_path in sorted(root.glob("*/scenario_manifest.json")):
        try:
            data = json.loads(manifest_path.read_text(encoding="utf-8"))
        except json.JSONDecodeError as exc:
            data = {
                "scenario": manifest_path.parent.name,
                "parse_error": str(exc),
                "requirement_ids": [],
                "tag": "unknown",
            }
        data["_evidence_dir"] = manifest_path.parent
        manifests.append(data)
    return manifests


def _read_json_if_exists(path: pathlib.Path) -> Optional[Any]:
    if not path.exists():
        return None
    try:
        return json.loads(path.read_text(encoding="utf-8"))
    except json.JSONDecodeError:
        return None


def _compute_recorded_pnl(execution_orders: Optional[List[Dict[str, Any]]]) -> Optional[float]:
    """Sum realized PnL across execution_orders for the scenario. Returns None if no data."""
    if not execution_orders:
        return None
    total = 0.0
    for row in execution_orders:
        # Columns are approximate; inspect actual schema. Common fields:
        # - fill_price, fill_qty, side, fee (per execution_orders schema in Phase 2)
        # - realized_pnl (if directly stored)
        if "realized_pnl" in row and row["realized_pnl"] is not None:
            try:
                total += float(row["realized_pnl"])
            except (TypeError, ValueError):
                continue
    return total if total != 0.0 else None


def reconcile_pnl_across_manifests(
    manifests: List[Dict[str, Any]],
    tolerance: float = RECONCILE_TOLERANCE_USD,
) -> ReconcileReport:
    """For each manifest, inspect pre/post balance snapshots (if present) and compute discrepancy.
    Real-tagged scenarios with balance snapshots are hard-gated; injected scenarios pass reconciliation trivially."""
    report = ReconcileReport(tolerance=tolerance)

    for m in manifests:
        evidence_dir = m["_evidence_dir"]
        scenario = m.get("scenario", evidence_dir.name)
        tag = m.get("tag", "unknown")
        req_ids = m.get("requirement_ids", [])

        # PnL reconciliation (TEST-03).
        pnl_check_applicable = False
        pnl_delta = pnl_recorded = pnl_discrepancy = None
        pnl_passed = None

        pre = _read_json_if_exists(evidence_dir / "balances_pre.json")
        post = _read_json_if_exists(evidence_dir / "balances_post.json")
        exec_orders = _read_json_if_exists(evidence_dir / "execution_orders.json")

        if pre and post:
            pnl_check_applicable = True
            # Pick the relevant platform from manifest context (scenario name heuristic).
            if "kalshi" in scenario.lower():
                platform = "kalshi"
            elif "polymarket" in scenario.lower():
                platform = "polymarket"
            else:
                platform = None

            if platform and platform in pre and platform in post:
                pre_b = pre[platform].get("balance")
                post_b = post[platform].get("balance")
                if pre_b is not None and post_b is not None:
                    pnl_delta = float(post_b) - float(pre_b)
                    pnl_recorded = _compute_recorded_pnl(exec_orders) or 0.0
                    pnl_discrepancy = pnl_delta - pnl_recorded
                    pnl_passed = abs(pnl_discrepancy) <= tolerance

        # Fee reconciliation (TEST-04). Manifests for happy-path scenarios include platform_fee + computed_fee.
        fee_check_applicable = False
        fee_platform = fee_computed = fee_discrepancy = None
        fee_passed = None

        if "platform_fee" in m and "computed_fee" in m and m["platform_fee"] is not None:
            fee_check_applicable = True
            fee_platform = float(m["platform_fee"])
            fee_computed = float(m["computed_fee"])
            fee_discrepancy = fee_platform - fee_computed
            fee_passed = abs(fee_discrepancy) <= tolerance

        # Overall.
        # - If neither check applies (injected, or no balance/fee data), overall = manifest's own status claim
        #   (if status field exists and is PASS/FILLED, count as pass).
        # - If both apply, both must pass.
        # - If only one applies, that one must pass.
        applicable_checks = [c for c in [pnl_passed, fee_passed] if c is not None]
        if applicable_checks:
            overall_passed = all(applicable_checks)
        else:
            # No reconciliation data. For injected scenarios, trust the scenario's own pass/fail.
            # For real scenarios without balance/fee data (e.g., FOK rejection, timeout-cancel), same.
            # Use manifest fields that signal pass: exec_01_invariant_holds, cancel_succeeded, or status != FAILED in a way contextually.
            overall_passed = (
                bool(m.get("exec_01_invariant_holds", True))
                and bool(m.get("cancelled_on_platform", True))
                and bool(m.get("cancel_succeeded", True))
                and bool(m.get("cr_02_lookup_succeeded", True))
                and bool(m.get("order_cancelled_on_platform", True))
            )

        result = ScenarioReconcileResult(
            scenario=scenario,
            evidence_dir=evidence_dir,
            requirement_ids=req_ids,
            tag=tag,
            pnl_check_applicable=pnl_check_applicable,
            pnl_delta=pnl_delta,
            pnl_recorded=pnl_recorded,
            pnl_discrepancy=pnl_discrepancy,
            pnl_passed=pnl_passed,
            fee_check_applicable=fee_check_applicable,
            fee_platform=fee_platform,
            fee_computed=fee_computed,
            fee_discrepancy=fee_discrepancy,
            fee_passed=fee_passed,
            overall_passed=overall_passed,
        )
        report.results.append(result)

    return report


def render_validation_markdown(
    manifests: List[Dict[str, Any]],

    """Generate the final 04-VALIDATION.md content. Based on 03-VERIFICATION.md structure."""
    now_iso = datetime.now(timezone.utc).strftime("%Y-%m-%dT%H:%M:%SZ")
    status_emoji_pass = "PASS"
    status_emoji_fail = "FAIL"

    lines: List[str] = []
    # Frontmatter
    lines.append("---")
    lines.append("phase: 4")
    lines.append("slug: sandbox-validation")
    lines.append(f"status: {'complete' if report.phase_gate_status == 'PASS' else 'incomplete'}")
    lines.append(f"phase_gate_status: {report.phase_gate_status}")
    lines.append("nyquist_compliant: true")
    lines.append("wave_0_complete: true")
    lines.append(f"generated: {now_iso}")
    lines.append(f"tolerance_usd: {report.tolerance}")
    lines.append(f"total_scenarios: {len(report.results)}")
    passed_count = sum(1 for r in report.results if r.overall_passed)
    lines.append(f"scenarios_passed: {passed_count}")
    lines.append(f"scenarios_failed: {len(report.results) - passed_count}")
    lines.append("---")
    lines.append("")

    # Title + header.
    lines.append("# Phase 4: Sandbox Validation — Acceptance Report")
    lines.append("")
    lines.append("**Phase Goal:** The full pipeline (collect -> scan -> execute -> monitor -> reconcile) is validated end-to-end against real platform APIs in sandbox/demo mode with no real money at risk.")
    lines.append(f"**Generated:** {now_iso}")
    lines.append(f"**Phase Gate Status:** **{report.phase_gate_status}** — Phase 5 {'UNBLOCKED' if report.phase_gate_status == 'PASS' else 'BLOCKED per D-19'}")
    lines.append(f"**Tolerance:** +/-${report.tolerance:.2f} absolute (D-17)")

    # Per-scenario table.

async def test_phase_reconciliation_and_validation_report():
    """Collect all scenario manifests, apply reconciliation tolerance, write 04-VALIDATION.md, hard-gate on breach."""
    manifests = collect_scenario_manifests()
    assert manifests, (
        "No scenario manifests found under evidence/04/. This terminal test requires that Plans "
        "04-03 through 04-07 have been run live (pytest -m live --live) before this test. "
        "Expected >=9 manifests (one per scenario)."
    )

    report = reconcile_pnl_across_manifests(manifests)

    # ALWAYS write the VALIDATION.md so operator has the full report even on failure.
    write_validation_markdown(manifests, report, target_path=VALIDATION_MD_PATH)

    # D-19 hard gate: any real-tagged scenario with a tolerance breach -> FAIL the test suite.
    if report.any_real_breach:
        breach_lines = []
        for r in report.results:
            if r.tag == "real" and not r.overall_passed:
                breach_lines.append(
                    f"- {r.scenario}: pnl_discrepancy={r.pnl_discrepancy} fee_discrepancy={r.fee_discrepancy}"
                )
        pytest.fail(
            "D-19 HARD GATE: Phase 4 has reconciliation breach on real-tagged scenarios. Phase 5 BLOCKED.\n"
            + "\n".join(breach_lines)
            + f"\nSee: {VALIDATION_MD_PATH}"
        )

    # Scenario-count sanity: expect 9 scenarios for full phase coverage (operator may run a subset for debugging).
    # Warning-only: not a hard gate.
    expected_scenarios = {
