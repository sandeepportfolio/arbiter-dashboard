```
"""Kill-switch live-fire against Kalshi demo (Scenario 6: SAFE-01)."""
from __future__ import annotations

import os
import json
import asyncio
import time
from unittest.mock import AsyncMock, MagicMock

import pytest
import structlog

from arbiter.sandbox import evidence

log = structlog.get_logger("arbiter.sandbox.killswitch")

# Operator-supplied from Task 0.
KS_MARKET_TICKER = "REPLACE-WITH-OPERATOR-SUPPLIED-TICKER"
KS_RESTING_PRICE = 0.05
KS_RESTING_QTY = 3


def _build_supervisor_with_real_adapter(kalshi_adapter):
    """Construct a SafetySupervisor around the REAL demo Kalshi adapter.
    Mock the notifier and config as minimally as needed for trip_kill to run."""
    from arbiter.safety.supervisor import SafetySupervisor

    notifier = AsyncMock()
    notifier.send = AsyncMock(return_value=None)

    # Minimal SafetyConfig — inspect the real dataclass in arbiter/safety/config.py (or wherever)
    # and build a real SafetyConfig with test values. If a factory exists in safety/test_supervisor.py
    # (e.g., _make_config or similar), use it. Otherwise, use a MagicMock with required attrs.
    config = MagicMock()
    config.min_cooldown_seconds = 0  # allow immediate reset if needed
    config.max_platform_exposure_usd = 1_000_000
    # Map of platform -> adapter
    adapters = {"kalshi": kalshi_adapter}

    # Construct using the actual signature from arbiter/safety/test_supervisor.py::_build_supervisor
    # (Read that helper and adapt — likely: SafetySupervisor(adapters=adapters, notifier=notifier, config=config, ...))
    supervisor = SafetySupervisor(adapters=adapters, notifier=notifier, config=config)
    return supervisor, notifier


@pytest.mark.live
async def test_kill_switch_cancels_open_kalshi_demo_order(
    demo_kalshi_adapter, sandbox_db_pool, evidence_dir,
):
    """Place resting Kalshi demo order, trip kill switch, assert cancellation on platform within 5s."""
    adapter = demo_kalshi_adapter
    assert "arbiter_sandbox" in os.getenv("DATABASE_URL", ""), "wrong DB"

    from arbiter.execution.engine import OrderStatus  # adjust path if needed

    supervisor, notifier = _build_supervisor_with_real_adapter(adapter)
    queue = supervisor.subscribe()

    # Step 1: place resting (non-FOK) order. See Plan 04-03 Task 3 for the pattern — may use
    # adapter.place_resting_limit or adapter._client.create_order with time_in_force=GTC.
    arb_id = "ARB-SANDBOX-KILLSWITCH"
    client_order_id = f"{arb_id}-YES-{os.urandom(4).hex()}"
    resting = await adapter.place_resting_limit(
        arb_id=arb_id,
        market_id=KS_MARKET_TICKER,
        canonical_id=KS_MARKET_TICKER,
        side="yes",
        price=KS_RESTING_PRICE,
        qty=KS_RESTING_QTY,
        client_order_id=client_order_id,
    )
    log.info("scenario.killswitch.resting_placed", order_id=resting.order_id, client_order_id=client_order_id)

    # Settle briefly so the order is actually resting on the book before we trip kill.
    await asyncio.sleep(1.0)

    # Step 2: trip the kill switch. 5s budget per SAFE-01.
    t_start = time.time()
    state = await asyncio.wait_for(
        supervisor.trip_kill(by="operator:phase4_test", reason="Phase 4 SAFE-01 live validation"),
        timeout=6.0,  # 5s SAFE-01 budget + 1s grace for network jitter
    )
    t_elapsed = time.time() - t_start
    assert state.armed is True, "trip_kill did not arm the supervisor"
    assert t_elapsed < 5.5, f"trip_kill exceeded 5s budget: took {t_elapsed:.2f}s"

    log.info("scenario.killswitch.tripped", elapsed_s=t_elapsed, state=str(state.to_dict() if hasattr(state, "to_dict") else state))

    # Step 3: confirm WS event was published.
    try:
        event = queue.get_nowait()
    except asyncio.QueueEmpty:
        pytest.fail("SafetySupervisor did not publish kill_switch event on trip_kill")
    assert event.get("type") == "kill_switch", f"Unexpected WS event type: {event.get('type')}"

    # Step 4: confirm platform actually cancelled the order.
    # Query demo Kalshi for the order's final state.
    final = await adapter.get_order(resting.order_id)
    # Accept either OrderStatus.CANCELLED or missing (if demo drops cancelled orders from queryable set).
    cancelled_on_platform = (final is None) or (final.status == OrderStatus.CANCELLED)
    assert cancelled_on_platform, (
        f"SAFE-01 INVARIANT VIOLATED: order {resting.order_id} still shows status={getattr(final, 'status', 'unknown')} "
        f"on demo exchange after trip_kill. adapter.cancel_all did not actually cancel on the platform."
    )

    # Step 5: confirm supervisor rejects new execution attempts while armed.
    # Try to place another FOK via adapter directly — adapter does NOT consult supervisor; engine does.
    # For this test, we assert via supervisor.allow_execution which returns (False, reason, state) while armed.
    allowed, reason, sup_state = supervisor.allow_execution(
        opportunity=MagicMock(),  # placeholder — allow_execution may only care about armed state
    )
    assert not allowed, "supervisor.allow_execution should be False while armed"
    assert "Kill switch armed" in reason or "armed" in reason.lower(), (
        f"Expected armed-state rejection reason; got: {reason!r}"
    )

    log.info("scenario.killswitch.allow_execution_rejected", reason=reason)

    # Evidence.
    await evidence.dump_execution_tables(sandbox_db_pool, evidence_dir)
    (evidence_dir / "scenario_manifest.json").write_text(json.dumps({
        "scenario": "kill_switch_cancels_open_kalshi_demo_order",
        "requirement_ids": ["SAFE-01", "TEST-01"],
        "phase_3_refs": ["03-01-PLAN", "03-HUMAN-UAT.md Test 1 (partial — backend only; UI reserved)"],
        "tag": "real",
        "placed_order_id": resting.order_id,
        "placed_client_order_id": client_order_id,
        "market": KS_MARKET_TICKER,
        "price": KS_RESTING_PRICE,
        "qty": KS_RESTING_QTY,
        "trip_kill_elapsed_s": t_elapsed,
        "cancelled_on_platform": cancelled_on_platform,
        "ws_event_type": event.get("type"),
        "supervisor_armed_post_trip": bool(state.armed),
        "allow_execution_rejected_while_armed": not allowed,
        "rejection_reason": reason,
    }, indent=2), encoding="utf-8")

    # Teardown hint: the supervisor is in armed state. Next scenario should either use a fresh supervisor
    # (fixture-scoped "function") or call supervisor.reset_kill() here. Because this is the only test
    # in this file and fixtures are function-scoped, we accept leaving it armed — pytest fixture teardown
    # tears down the adapter.
```
