```
"""Kalshi demo happy-path lifecycle (Scenario 1: TEST-01 + TEST-04)."""
from __future__ import annotations

import os
import json
from decimal import Decimal

import pytest
import structlog

from arbiter.sandbox import evidence, reconcile
from arbiter.config.settings import kalshi_order_fee

log = structlog.get_logger("arbiter.sandbox.kalshi_happy")

# TARGET MARKET — replace with ticker from Task 0 operator pre-flight.
# Happy-path market must have: liquid book (>=10 contracts depth at target price),
# price under ~$0.60 so notional qty*price stays small, demo-tradeable.
HAPPY_MARKET_TICKER = "REPLACE-WITH-OPERATOR-SUPPLIED-TICKER"
HAPPY_TARGET_PRICE = 0.50  # adjust based on market mid; operator may override
HAPPY_TARGET_QTY = 5       # small test size; notional = 5 * 0.50 = $2.50


@pytest.mark.live
async def test_kalshi_happy_lifecycle(
    demo_kalshi_adapter, sandbox_db_pool, evidence_dir, balance_snapshot,
):
    """Submit FOK on liquid demo market -> fill -> DB row -> evidence dump -> fee_cost assertion."""
    adapter = demo_kalshi_adapter

    # Guard: the fixture already asserts KALSHI_BASE_URL contains demo-api.kalshi.co.
    # Second check: ensure DATABASE_URL is sandbox so execution_* writes go there.
    assert "arbiter_sandbox" in os.getenv("DATABASE_URL", ""), "wrong DB"

    # Pre-balance snapshot (TEST-03 input).
    pre_balances = await balance_snapshot()

    # Place FOK at mid.
    arb_id = "ARB-SANDBOX-KALSHI-HAPPY"
    order = await adapter.place_fok(
        arb_id=arb_id,
        market_id=HAPPY_MARKET_TICKER,
        canonical_id=HAPPY_MARKET_TICKER,  # for single-platform, canonical == market
        side="yes",
        price=HAPPY_TARGET_PRICE,
        qty=HAPPY_TARGET_QTY,
    )

    log.info(
        "scenario.kalshi_happy.order_returned",
        arb_id=arb_id, order_id=order.order_id, status=str(order.status),
        external_client_order_id=order.external_client_order_id,
    )

    # TEST-01 assertion: full lifecycle produced a FILLED order.
    from arbiter.execution.engine import OrderStatus  # or arbiter.execution.types — use what adapter file uses
    assert order.status == OrderStatus.FILLED, (
        f"Expected FILLED on liquid demo market {HAPPY_MARKET_TICKER}; got {order.status}. "
        f"Error: {getattr(order, 'error', None)}. "
        f"Possible causes: market changed (thin liquidity now), price moved past limit, demo account not funded."
    )

    # TEST-04 fee assertion: pull fill from /portfolio/fills and read fee_cost (NOT realized_fee — Pitfall 1).
    # The exact method on KalshiAdapter for fetching fills may be named get_fills, fetch_fills,
    # or accessible via adapter._client.get_fills. Inspect arbiter/execution/adapters/kalshi.py.
    fills = await adapter.get_fills(order_id=order.order_id)
    assert fills, f"No fills returned for FILLED order {order.order_id}"
    first_fill = fills[0]
    # fee_cost is a fixed-point DOLLAR STRING per Kalshi Jan 2026 API migration.
    platform_fee = float(Decimal(str(first_fill["fee_cost"])))
    computed_fee = kalshi_order_fee(HAPPY_TARGET_PRICE, HAPPY_TARGET_QTY)
    reconcile.assert_fee_matches("kalshi", platform_fee, computed_fee)

    log.info(
        "scenario.kalshi_happy.fee_match",
        platform_fee=platform_fee, computed_fee=computed_fee,
        delta=abs(platform_fee - computed_fee),
    )

    # Post-balance snapshot (TEST-03 input; aggregator in Plan 04-08 does the hard-gate assertion).
    post_balances = await balance_snapshot()

    # Evidence capture.
    await evidence.dump_execution_tables(sandbox_db_pool, evidence_dir)
    evidence.write_balances(evidence_dir, pre_balances, post_balances)
    # Emit a scenario-manifest JSON for aggregator consumption.
    (evidence_dir / "scenario_manifest.json").write_text(json.dumps({
        "scenario": "kalshi_happy_lifecycle",
        "requirement_ids": ["TEST-01", "TEST-04"],
        "tag": "real",
        "order_id": order.order_id,
        "external_client_order_id": order.external_client_order_id,
        "platform_fee": platform_fee,
        "computed_fee": computed_fee,
        "market": HAPPY_MARKET_TICKER,
        "side": "yes",
        "price": HAPPY_TARGET_PRICE,
        "qty": HAPPY_TARGET_QTY,
        "status": str(order.status),
    }, indent=2), encoding="utf-8")
```
