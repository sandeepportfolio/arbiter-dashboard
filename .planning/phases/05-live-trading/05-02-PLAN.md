    ```python
    """Phase 5 auto-abort primitive — wire reconcile_post_trade to SafetySupervisor.trip_kill on breach.

    Fail-closed: if reconcile itself raises, the kill switch STILL trips. Per research
    §Security Domain "Reconciliation step itself fails silently -> default to abort (fail-closed)."
    """
    from __future__ import annotations
    import structlog
    from typing import Any, Callable, Dict, List, Optional

    log = structlog.get_logger("arbiter.live.auto_abort")

    async def wire_auto_abort_on_reconcile(
        supervisor,                              # SafetySupervisor
        reconcile_fn: Callable[[], Any],        # async callable returning List[Dict] or None
    ) -> Dict[str, Any]:
        """Run reconcile; on non-empty discrepancy list OR any exception, trip kill-switch.

        Returns:
            {"aborted": bool, "discrepancies": List[Dict], "error": Optional[str]}
        """
        try:
            discrepancies = await reconcile_fn()
        except Exception as exc:
            log.error("phase5.reconcile.exception", err=str(exc))
            reason = f"reconcile_error: {type(exc).__name__}: {exc}"
            try:
                await supervisor.trip_kill(by="system:phase5_reconcile_fail", reason=reason)
            except Exception as trip_exc:
                log.critical("phase5.trip_kill.failed", err=str(trip_exc))
            return {"aborted": True, "discrepancies": [], "error": str(exc)}

        if not discrepancies:  # None or empty list
            log.info("phase5.reconcile.ok")
            return {"aborted": False, "discrepancies": []}

        # Non-empty discrepancy list -> auto-abort
        reason = _format_reason(discrepancies)
        log.warning("phase5.reconcile.breach", discrepancies=discrepancies)
        try:
            await supervisor.trip_kill(by="system:phase5_reconcile_fail", reason=reason)
        except Exception as trip_exc:
            log.critical("phase5.trip_kill.failed", err=str(trip_exc))
        return {"aborted": True, "discrepancies": list(discrepancies), "error": None}

    def _format_reason(discrepancies: List[Dict[str, Any]]) -> str:
        """Build a kill-switch reason string that includes each breach's shape."""
        parts = []
        for d in discrepancies[:3]:  # cap at 3 for readability
            kind = d.get("reason", "unknown")
            amt = d.get("discrepancy", 0.0)
            plat = d.get("platform", "?")
            parts.append(f"{plat}:{kind}={amt:+.4f}")
        return "phase5_reconcile_fail: " + "; ".join(parts)
    ```
