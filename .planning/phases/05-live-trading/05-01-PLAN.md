    ```python
    from __future__ import annotations

    from arbiter.sandbox.reconcile import (

    )

        "RECONCILE_TOLERANCE_USD",
        "assert_pnl_within_tolerance",
        "assert_fee_matches",
        "reconcile_post_trade",
    ]

    async def reconcile_post_trade(
        execution,                      # ArbExecution
        adapters: Dict[str, Any],       # {"kalshi": KalshiAdapter, "polymarket": PolymarketAdapter}
        tolerance: float = RECONCILE_TOLERANCE_USD,
        fee_fetcher: Callable = None,   # optional async (platform, order_id) -> float for unit testing
    ) -> List[Dict[str, Any]]:
        """
        Post-trade reconciliation: compare recorded fee/fill_price against platform-reported.
        Returns list of discrepancy dicts — empty list means all legs reconciled.
        Does NOT call trip_kill — caller (Plan 05-02) wires the auto-abort.
        """
        from arbiter.execution.engine import OrderStatus
        discrepancies: List[Dict[str, Any]] = []
        for leg in (execution.leg_yes, execution.leg_no):
            if leg.status != OrderStatus.FILLED:
                continue
            # Fetch platform-reported fee via adapter-specific hook.
            # For now, the adapter hook is called optionally — Plan 05-02 provides the real impl.
            if fee_fetcher is None:
                continue  # no discrepancy possible without ground-truth source
            platform_fee = await fee_fetcher(leg.platform, leg.order_id)
            # Compute expected fee from fee model.
            from arbiter.config.settings import kalshi_order_fee, polymarket_order_fee
            if leg.platform == "kalshi":
                computed_fee = kalshi_order_fee(leg.fill_price, leg.fill_qty)
            elif leg.platform == "polymarket":
                computed_fee = polymarket_order_fee(leg.canonical_id, leg.fill_price, leg.fill_qty)
            else:
                continue
            if abs(platform_fee - computed_fee) > tolerance:
                discrepancies.append({
                    "arb_id": execution.arb_id,
                    "leg_order_id": leg.order_id,
                    "platform": leg.platform,
                    "platform_fee": platform_fee,
                    "computed_fee": computed_fee,
                    "discrepancy": platform_fee - computed_fee,
                    "tolerance": tolerance,
                    "reason": "fee_mismatch",
                })
        return discrepancies
    ```
